### 1. Component-Based Architecture

Modern web applications benefit greatly from a component-based approach. This means:

- **Reusability**: Components can be reused across different parts of your application
- **Maintainability**: Easier to update and debug individual components
- **Testability**: Components can be tested in isolation
- **Collaboration**: Team members can work on different components simultaneously

```jsx title="components/Button.tsx" showLineNumbers
// Example of a reusable Button component
interface ButtonProps {
  variant: 'primary' | 'secondary'
  children: React.ReactNode
  onClick?: () => void
  disabled?: boolean
}

const Button: React.FC<ButtonProps> = ({ 
  variant, 
  children, 
  onClick, 
  disabled = false,
  ...props 
}) => {
  const baseClasses = "px-4 py-2 rounded-md font-medium transition-colors focus:outline-none focus:ring-2 focus:ring-offset-2"
  const variantClasses = {
    primary: "bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500 disabled:bg-blue-300",
    secondary: "bg-gray-200 text-gray-800 hover:bg-gray-300 focus:ring-gray-500 disabled:bg-gray-100"
  }
  
  return (
    <button 
      className={`${baseClasses} ${variantClasses[variant]} ${disabled ? 'cursor-not-allowed' : 'cursor-pointer'}`}
      onClick={onClick}
      disabled={disabled}
      {...props}
    >
      {children}
    </button>
  )
}

export default Button
```

### 1. TypeScript Integration

Using TypeScript provides numerous benefits:

```typescript title="types/user.ts" showLineNumbers
// Define comprehensive user types
interface User {
  id: string
  name: string
  email: string
  role: 'admin' | 'user' | 'moderator'
  avatar?: string
  createdAt: Date
  lastLoginAt?: Date
  isActive: boolean
}

interface UserProfileProps {
  user: User
  showEmail?: boolean
  onEdit?: (user: User) => void
}

const UserProfile: React.FC<UserProfileProps> = ({ 
  user, 
  showEmail = false, 
  onEdit 
}) => {
  const getRoleColor = (role: User['role']) => {
    const colors = {
      admin: 'bg-red-100 text-red-800',
      moderator: 'bg-yellow-100 text-yellow-800', 
      user: 'bg-green-100 text-green-800'
    }
    return colors[role]
  }

  return (
    <div className="user-profile bg-white dark:bg-gray-800 rounded-lg p-6 shadow-sm">
      <div className="flex items-center space-x-4">
        {user.avatar && (
          <img 
            src={user.avatar} 
            alt={`${user.name}'s avatar`}
            className="w-16 h-16 rounded-full object-cover"
          />
        )}
        <div>
          <h2 className="text-xl font-semibold text-gray-900 dark:text-white">
            {user.name}
          </h2>
          {showEmail && (
            <p className="text-gray-600 dark:text-gray-300">{user.email}</p>
          )}
          <span className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${getRoleColor(user.role)}`}>
            {user.role}
          </span>
        </div>
      </div>
      {onEdit && (
        <button 
          onClick={() => onEdit(user)}
          className="mt-4 px-4 py-2 bg-blue-600 text-white rounded-md hover:bg-blue-700 transition-colors"
        >
          Edit Profile
        </button>
      )}
    </div>
  )
}

export default UserProfile
```

<Callout type="info">
üí° **TypeScript Tip**: Use union types for enum-like values and always define comprehensive interfaces for better development experience.
</Callout>

```typescript title="middleware/security.ts" showLineNumbers
// Example security middleware for Next.js
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export function middleware(request: NextRequest) {
  const response = NextResponse.next()
  
  // Security headers
  response.headers.set('X-Frame-Options', 'DENY')
  response.headers.set('X-Content-Type-Options', 'nosniff')
  response.headers.set('Referrer-Policy', 'origin-when-cross-origin')
  response.headers.set(
    'Content-Security-Policy',
    "default-src 'self'; script-src 'self' 'unsafe-eval'"
  )
  
  return response
}
```
